-- 1. Top performing regions this month
SELECT
    region,
    SUM(total_revenue)    AS monthly_revenue,
    SUM(order_count)      AS monthly_orders,
    AVG(avg_order_value)  AS avg_basket_size,
    RANK() OVER (ORDER BY SUM(total_revenue) DESC) AS revenue_rank
FROM dbo.daily_sales_summary
WHERE order_date >= DATEADD(month, DATEDIFF(month, 0, GETDATE()), 0)
GROUP BY region
ORDER BY monthly_revenue DESC;

-- 2. Week-over-week revenue growth by category
WITH weekly AS (
    SELECT
        DATEPART(week, order_date) AS week_num,
        product_category,
        SUM(total_revenue) AS weekly_revenue
    FROM dbo.daily_sales_summary
    GROUP BY DATEPART(week, order_date), product_category
)
SELECT
    week_num,
    product_category,
    weekly_revenue,
    LAG(weekly_revenue) OVER (PARTITION BY product_category ORDER BY week_num) AS prev_week_revenue,
    ROUND(
        (weekly_revenue - LAG(weekly_revenue) OVER (PARTITION BY product_category ORDER BY week_num))
        / NULLIF(LAG(weekly_revenue) OVER (PARTITION BY product_category ORDER BY week_num), 0) * 100,
        2
    ) AS wow_growth_pct
FROM weekly
