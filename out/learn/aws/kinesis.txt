# Example: website clickstream events sent as they happen

import boto3
import json
from datetime import datetime
import random

kinesis = boto3.client('kinesis', region_name='us-east-1')
STREAM_NAME = 'clickstream-events'

def send_click_event(user_id: str, page: str, action: str):
    event = {
        "event_id":  f"evt_{random.randint(100000, 999999)}",
        "user_id":   user_id,
        "page":      page,
        "action":    action,
        "timestamp": datetime.utcnow().isoformat(),
        "session_id": f"sess_{user_id[:8]}"
    }
    
    kinesis.put_record(
        StreamName=STREAM_NAME,
        Data=json.dumps(event),
        PartitionKey=user_id  # Same user's events always go to same shard
    )

# Send a batch of events (more efficient than one at a time)
def send_batch(events: list):
    records = [
        {"Data": json.dumps(e), "PartitionKey": e["user_id"]}
        for e in events
    ]
    kinesis.put_records(StreamName=STREAM_NAME, Records=records)

# Simulate 100 click events
for i in range(100):
    send_click_event(f"user_{i}", "/products/laptop", "view")
