# This runs serverlessly — no cluster management needed

import sys
from awsglue.transforms import *
from awsglue.utils import getResolvedOptions
from pyspark.context import SparkContext
from awsglue.context import GlueContext
from awsglue.job import Job
from pyspark.sql import functions as F

# Initialize Glue context
args = getResolvedOptions(sys.argv, ['JOB_NAME', 'run_date'])
sc = SparkContext()
glueContext = GlueContext(sc)
spark = glueContext.spark_session
job = Job(glueContext)
job.init(args['JOB_NAME'], args)

run_date = args['run_date']  # Passed from Step Functions

# Read from S3 Bronze using Glue DynamicFrame
bronze_dyf = glueContext.create_dynamic_frame.from_options(
    format_options={"withHeader": True, "separator": ","},
    connection_type="s3",
    format="csv",
    connection_options={"paths": [f"s3://yourcompany-datalake-bronze/sales/{run_date}/"]},
)

# Convert to Spark DataFrame for transformations
df = bronze_dyf.toDF()

# Data quality and transformation
df_clean = (df
    .dropDuplicates(["order_id"])
    .filter(F.col("order_id").isNotNull())
    .withColumn("order_date",   F.to_date("order_date", "yyyy-MM-dd"))
    .withColumn("unit_price",   F.col("unit_price").cast("double"))
    .withColumn("quantity",     F.col("quantity").cast("integer"))
    .withColumn("net_amount",   F.col("unit_price") * F.col("quantity"))
    .withColumn("_load_date",   F.lit(run_date))
)

# Write to S3 Silver as Parquet
df_clean.write.mode("append").partitionBy("order_date").parquet(
    "s3://yourcompany-datalake-silver/sales/"
)

print(f"✅ Glue job complete: {df_clean.count()} rows written to Silver")
